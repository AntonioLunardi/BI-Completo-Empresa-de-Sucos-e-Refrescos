SET [MARCAS] AS
'ORDER([Produto].[Hierarquia Produto].[Nível Marca], [Measures].[Margem], BDESC)'

'[Measures].[Margem]', FORMAT_STRING = '##,###.00'

'RANK([Produto].[Hierarquia Produto].Currentmember, [MARCAS])', FORMAT_STRING = "#;#;-"

({[MARCAS]}) On Rows
FROM [COMPLETO] WHERE ([Tempo].[Ano].&[2014])
